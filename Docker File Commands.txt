Application Properties
spring.application.name=Invoice
# Database URL
spring.datasource.url=jdbc:mysql://${MYSQL_HOST:localhost}:${MYSQL_PORT:3306}/${MYSQL_DB_NAME:invoice_payment}

# Database Credentials
spring.datasource.username=${MYSQL_USER:root}
spring.datasource.password=${MYSQL_PASSWORD: }

# JDBC Driver
spring.datasource.driver-class-name=com.mysql.cj.jdbc.Driver

# Hibernate (JPA) Properties
spring.jpa.hibernate.ddl-auto=update
spring.jpa.show-sql=true
spring.jpa.properties.hibernate.dialect=org.hibernate.dialect.MySQL8Dialect



To Create Project Image:
docker build -t invoice_payment .

To Run the Spring Boot Application :
docker run -p 8090:8080 --name invoice_payment --net networkmysql -e MYSQL_HOST=mysqlcontainer -e MYSQL_PORT=3307 -e MYSQL_DB_NAME=invoice_payment -e MYSQL_USER=root -e MYSQL_PASSWORD=root invoice_payment
